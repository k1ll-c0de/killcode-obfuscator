private static void parseInput(String input) {
    JarFile jarFile = null;
    try {
        jarFile = new JarFile(input);
    }
    catch (IOException e) {
        e.printStackTrace();
    }
    JarFile finalJarFile = jarFile;
    jarFile.stream().forEach(entry -> {
        try {
            if (entry.getName().endsWith(".class")) {
                ClassReader classReader = new ClassReader(finalJarFile.getInputStream((ZipEntry)entry));
                ClassNode classNode = new ClassNode();
                classReader.accept(classNode, 2);
                classes.put(classNode.name, classNode);
            } else if (!entry.isDirectory()) {
                outputStream.putNextEntry(new ZipEntry(entry.getName()));
                outputStream.write(OutputUtil.toByteArray(finalJarFile.getInputStream((ZipEntry)entry)));
                outputStream.closeEntry();
            }
        }
        catch (Exception e) {
            e.printStackTrace();
        }
    });
    try {
        jarFile.close();
    }
    catch (IOException e) {
        e.printStackTrace();
    }
}
